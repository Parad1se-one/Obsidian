### Q1: LLM 的上下文和长短期记忆是怎么做的？

### 上下文管理架构

```python
class LLMContextManager:
    """LLM 上下文与记忆管理系统"""
    
    def __init__(self):
        # 短期记忆：最近 N 轮对话（完整保留）
        self.short_term_memory = deque(maxlen=20)
        
        # 长期记忆：向量数据库存储
        self.long_term_memory = VectorStore(backend="milvus")
        
        # 语义记忆：关键事实与用户画像
        self.semantic_memory = KeyValueStore(backend="redis")
        
        # 摘要记忆：对话摘要压缩
        self.summary_memory = ""
        
    def add_message(self, role: str, content: str):
        """添加消息到上下文"""
        # 1. 加入短期记忆
        self.short_term_memory.append({
            "role": role,
            "content": content,
            "timestamp": time.time()
        })
        
        # 2. 提取关键信息到长期记忆
        if len(content) > 200:
            self._extract_to_long_term(content)
        
        # 3. 更新摘要（每 10 轮）
        if len(self.short_term_memory) % 10 == 0:
            self._update_summary()
    
    def build_context(self, max_tokens: int = 4096) -> list:
        """构建模型上下文"""
        context = []
        token_count = 0
        
        # 1. 系统提示
        system_prompt = {"role": "system", "content": self._get_system_prompt()}
        context.append(system_prompt)
        token_count += self._count_tokens(system_prompt)
        
        # 2. 对话摘要（如果有）
        if self.summary_memory:
            summary_msg = {"role": "system", "content": f"对话摘要：{self.summary_memory}"}
            context.append(summary_msg)
            token_count += self._count_tokens(summary_msg)
        
        # 3. 最近对话（短期记忆）- 从后往前
        for msg in reversed(self.short_term_memory):
            if token_count + self._count_tokens(msg) > max_tokens * 0.7:
                break
            context.insert(2, msg)  # 插在 system 之后
            token_count += self._count_tokens(msg)
        
        # 4. 相关长期记忆（向量检索）
        if self.long_term_memory:
            last_msg = self.short_term_memory[-1]["content"] if self.short_term_memory else ""
            relevant = self.long_term_memory.search(query=last_msg, k=3)
            for mem in relevant:
                mem_msg = {"role": "system", "content": f"相关信息：{mem['content']}"}
                if token_count + self._count_tokens(mem_msg) > max_tokens:
                    break
                context.append(mem_msg)
                token_count += self._count_tokens(mem_msg)
        
        return context
    
    def _extract_to_long_term(self, content: str):
        """提取关键信息到长期记忆"""
        # 使用 LLM 提取关键实体和事件
        extracted = self.llm.extract_entities(content)
        
        # 向量化存储
        embedding = self.embedder.embed(content)
        self.long_term_memory.insert({
            "content": content,
            "entities": extracted,
            "embedding": embedding,
            "timestamp": time.time()
        })
    
    def _update_summary(self):
        """生成对话摘要"""
        conversation = "\n".join([
            f"{m['role']}: {m['content']}"
            for m in self.short_term_memory
        ])
        
        self.summary_memory = self.llm.generate(f"""
        请为以下对话生成简洁摘要（200 字以内）：
        
        {conversation}
        
        摘要：
        """)
```

### 长短期记忆对比

| 类型 | 存储内容 | 存储方式 | 访问速度 | 容量 |
|------|---------|---------|---------|------|
| **短期记忆** | 最近 10-20 轮对话 | 内存 (deque) | 毫秒级 | 有限 |
| **长期记忆** | 历史对话片段 | 向量数据库 | 秒级 | 大 |
| **语义记忆** | 用户画像/偏好/事实 | KV 存储 (Redis) | 毫秒级 | 中 |
| **摘要记忆** | 对话压缩摘要 | 字符串 | 毫秒级 | 小 |

### 关键技术点

1. **滑动窗口**：保留最近 N 轮，超出则丢弃或摘要
2. **向量检索**：用语义相似度检索相关历史
3. **摘要压缩**：定期用 LLM 生成对话摘要
4. **分层存储**：热数据在内存，冷数据在向量库

```python
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
import asyncio
import json

app = FastAPI()

async def generate_tokens(prompt: str):
    """模拟 LLM Token 生成"""
    tokens = ["你", "好", "，", "我", "是", "AI", "助手"]
    
    for token in tokens:
        # 模拟生成延迟
        await asyncio.sleep(0.1)
        
        # 发送 Token
        yield f"data: {json.dumps({'token': token})}\n\n"
    
    # 结束标记
    yield "data: [DONE]\n\n"

@app.post("/chat/stream")
async def chat_stream(request: dict):
    """流式聊天接口"""
    prompt = request.get("prompt", "")
    
    return StreamingResponse(
        generate_tokens(prompt),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
        }
    )
```

```javascript
import { useState } from 'react';

function ChatComponent() {
  const [response, setResponse] = useState('');
  const [loading, setLoading] = useState(false);

  const sendMessage = async (message) => {
    setLoading(true);
    setResponse('');

    const eventSource = new EventSource(`/chat/stream?prompt=${encodeURIComponent(message)}`);

    eventSource.onmessage = (event) => {
      if (event.data === '[DONE]') {
        eventSource.close();
        setLoading(false);
        return;
      }

      const data = JSON.parse(event.data);
      setResponse(prev => prev + data.token);
    };

    eventSource.onerror = () => {
      eventSource.close();
      setLoading(false);
    };
  };

  return (
    <div>
      <div className="response">{response}</div>
      <button onClick={() => sendMessage('你好')}>发送</button>
    </div>
  );
}
```

```python
class HighDimensionalAnalyzer:
    """高维数据影响因素分析"""
    
    def __init__(self):
        self.feature_importance = {}
        self.correlation_matrix = None
        
    def analyze(self, X: pd.DataFrame, y: pd.Series) -> dict:
        """分析影响因素"""
        results = {}
        
        # 1. 相关性分析
        results['correlation'] = self._correlation_analysis(X, y)
        
        # 2. 特征重要性（树模型）
        results['feature_importance'] = self._tree_importance(X, y)
        
        # 3. 降维可视化
        results['pca_components'] = self._pca_analysis(X)
        
        # 4. SHAP 值分析
        results['shap_values'] = self._shap_analysis(X, y)
        
        # 5. 因果推断
        results['causal_effect'] = self._causal_inference(X, y)
        
        return results
    
    def _correlation_analysis(self, X, y):
        """相关性分析"""
        corr = pd.DataFrame({
            'feature': X.columns,
            'correlation': [X[col].corr(y) for col in X.columns],
            'p_value': [stats.pearsonr(X[col], y)[1] for col in X.columns]
        })
        return corr.sort_values('correlation', key=abs, ascending=False)
    
    def _tree_importance(self, X, y):
        """基于树模型的特征重要性"""
        from sklearn.ensemble import RandomForestRegressor
        
        model = RandomForestRegressor(n_estimators=100, random_state=42)
        model.fit(X, y)
        
        importance = pd.DataFrame({
            'feature': X.columns,
            'importance': model.feature_importances_
        })
        return importance.sort_values('importance', ascending=False)
    
    def _shap_analysis(self, X, y):
        """SHAP 值分析（解释模型预测）"""
        import shap
        
        model = RandomForestRegressor()
        model.fit(X, y)
        
        explainer = shap.TreeExplainer(model)
        shap_values = explainer.shap_values(X)
        
        # 计算平均 SHAP 值（特征重要性）
        mean_shap = np.abs(shap_values).mean(axis=0)
        
        return pd.DataFrame({
            'feature': X.columns,
            'shap_importance': mean_shap
        }).sort_values('shap_importance', ascending=False)
```

### 分析方法对比

| 方法 | 适用场景 | 优点 | 缺点 |
|------|---------|------|------|
| **相关性分析** | 线性关系 | 简单快速 | 只能检测线性 |
| **树模型重要性** | 非线性关系 | 捕捉复杂交互 | 可能有偏 |
| **SHAP 值** | 模型解释 | 理论完备 | 计算量大 |
| **LASSO** | 特征选择 | 自动稀疏化 | 线性假设 |

```python
class CNN(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x)))
        x = x.view(-1, 64 * 7 * 7)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# RNN 实现
class RNN(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers):
        super().__init__()
        self.rnn = nn.RNN(input_size, hidden_size, num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_size, 10)
    
    def forward(self, x):
        out, _ = self.rnn(x)
        return self.fc(out[:, -1, :])

# LSTM 实现
class LSTM(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers):
        super().__init__()
        self.lstm = nn.LSTM(input_size, hidden_size, num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_size, 10)
    
    def forward(self, x):
        out, _ = self.lstm(x)
        return self.fc(out[:, -1, :])
```

### Q6: NLP 生成时如何选择最优序列？

### 解码策略对比

```python
class DecodingStrategies:
    """NLP 生成解码策略"""
    
    def greedy_decoding(self, logits):
        """贪婪解码：每步选概率最大"""
        return torch.argmax(logits, dim=-1)
    
    def beam_search(self, logits, beam_size=5):
        """束搜索：保留 K 个候选"""
        # 初始化
        beams = [(1.0, [])]  # (score, tokens)
        
        for step in range(max_length):
            candidates = []
            
            for score, tokens in beams:
                # 计算下一步概率
                probs = F.softmax(logits[step], dim=-1)
                top_k = torch.topk(probs, beam_size)
                
                for prob, token in zip(top_k.values, top_k.indices):
                    candidates.append((score * prob.item(), tokens + [token]))
            
            # 保留 Top-K
            beams = sorted(candidates, key=lambda x: x[0], reverse=True)[:beam_size]
        
        return beams[0][1]
    
    def top_k_sampling(self, logits, k=50):
        """Top-K 采样"""
        # 只保留概率最高的 K 个
        top_k_logits, top_k_indices = torch.topk(logits, k)
        
        # 重新归一化
        probs = F.softmax(top_k_logits, dim=-1)
        
        # 采样
        idx = torch.multinomial(probs, 1)
        return top_k_indices[idx]
    
    def top_p_sampling(self, logits, p=0.95):
        """Top-P (Nucleus) 采样"""
        # 排序
        sorted_logits, sorted_indices = torch.sort(logits, descending=True)
        sorted_probs = F.softmax(sorted_logits, dim=-1)
        
        # 累积概率
        cumulative_probs = torch.cumsum(sorted_probs, dim=-1)
        
        # 找到累积概率超过 p 的位置
        nucleus = cumulative_probs <= p
        
        # 截断
        sorted_logits[~nucleus] = -float('inf')
        
        # 采样
        probs = F.softmax(sorted_logits, dim=-1)
        idx = torch.multinomial(probs, 1)
        return sorted_indices[idx]
    
    def temperature_sampling(self, logits, temperature=0.7):
        """温度采样"""
        # 调整分布
        scaled_logits = logits / temperature
        
        # 采样
        probs = F.softmax(scaled_logits, dim=-1)
        return torch.multinomial(probs, 1)
```

```python
from collections import Counter
import heapq

def top_k_frequent_chars(s: str, k: int) -> list:
    """
    统计字符频率并返回 TopK
    
    Args:
        s: 输入字符串
        k: 返回前 K 个
    
    Returns:
        [(char, count), ...] 按频率降序
    """
    # 方法 1: Counter + most_common (最简单)
    counter = Counter(s)
    return counter.most_common(k)

def top_k_frequent_chars_heap(s: str, k: int) -> list:
    """使用最小堆（适合 k 远小于 n 的情况）"""
    counter = Counter(s)
    
    # 维护大小为 k 的最小堆
    heap = []
    for char, count in counter.items():
        if len(heap) < k:
            heapq.heappush(heap, (count, char))
        elif count > heap[0][0]:
            heapq.heapreplace(heap, (count, char))
    
    # 按频率降序返回
    return sorted(heap, key=lambda x: x[0], reverse=True)

# 测试
s = "hello world"
print(top_k_frequent_chars(s, 3))
# 输出：[('l', 3), ('o', 2), ('h', 1)]
```

### 手撕 2: LeetCode 153 旋转数组找最小值

```python
def findMin(nums: list[int]) -> int:
    """
    LeetCode 153: 寻找旋转排序数组中的最小值
    
    假设按照升序排序的数组在预先未知的某个点上进行了旋转。
    例如：[0,1,2,4,5,6,7] → [4,5,6,7,0,1,2]
    
    要求：O(log n) 时间复杂度 → 二分查找
    """
    left, right = 0, len(nums) - 1
    
    while left < right:
        mid = (left + right) // 2
        
        # 如果中间元素大于右端，最小值在右半部分
        if nums[mid] > nums[right]:
            left = mid + 1
        # 否则最小值在左半部分（包括 mid）
        else:
            right = mid
    
    return nums[left]

# 测试
print(findMin([3, 4, 5, 1, 2]))  # 输出：1
print(findMin([4, 5, 6, 7, 0, 1, 2]))  # 输出：0
print(findMin([11, 13, 15, 17]))  # 输出：11（未旋转）
```

### 变体：允许重复元素

```python
def findMinWithDuplicates(nums: list[int]) -> int:
    """
    LeetCode 154: 允许重复元素的旋转数组找最小值
    """
    left, right = 0, len(nums) - 1
    
    while left < right:
        mid = (left + right) // 2
        
        if nums[mid] > nums[right]:
            left = mid + 1
        elif nums[mid] < nums[right]:
            right = mid
        else:
            # nums[mid] == nums[right]，无法判断，缩小右边界
            right -= 1
    
    return nums[left]
```

### 手撕：有放回组合问题（n 选 k 组合数）

```python
def combinations_with_replacement(n: int, k: int) -> int:
    """
    有放回组合：从 n 个元素中选 k 个（可重复）
    
    公式：C(n+k-1, k) = (n+k-1)! / (k! * (n-1)!)
    """
    # 方法 1: 数学公式
    from math import comb
    return comb(n + k - 1, k)

def combinations_with_replacement_dp(n: int, k: int) -> int:
    """
    动态规划解法
    
    dp[i][j] = 从前 i 个元素中选 j 个的组合数
    
    状态转移：
    - 不选第 i 个：dp[i-1][j]
    - 选第 i 个：dp[i][j-1]（因为有放回，还可以继续选）
    
    dp[i][j] = dp[i-1][j] + dp[i][j-1]
    """
    # 初始化
    dp = [[0] * (k + 1) for _ in range(n + 1)]
    
    # 边界条件：选 0 个只有 1 种方式
    for i in range(n + 1):
        dp[i][0] = 1
    
    # 状态转移
    for i in range(1, n + 1):
        for j in range(1, k + 1):
            dp[i][j] = dp[i-1][j] + dp[i][j-1]
    
    return dp[n][k]

def combinations_with_replacement_recursive(n: int, k: int) -> int:
    """递归解法（带记忆化）"""
    from functools import lru_cache
    
    @lru_cache(maxsize=None)
    def dfs(remaining_n: int, remaining_k: int) -> int:
        # 边界条件
        if remaining_k == 0:
            return 1
        if remaining_n == 0:
            return 0
        
        # 两种选择：
        # 1. 不选当前元素
        # 2. 选当前元素（还可以继续选）
        return dfs(remaining_n - 1, remaining_k) + dfs(remaining_n, remaining_k - 1)
    
    return dfs(n, k)

# 测试
print(combinations_with_replacement(3, 2))  # 输出：6
# 解释：从{1,2,3}中选 2 个（可重复）：(1,1), (1,2), (1,3), (2,2), (2,3), (3,3)

print(combinations_with_replacement_dp(3, 2))  # 输出：6
print(combinations_with_replacement_recursive(3, 2))  # 输出：6
```

### 优化递归：使用 lru_cache

```python
from functools import lru_cache

@lru_cache(maxsize=None)
def combinations_optimized(n: int, k: int) -> int:
    """使用 lru_cache 优化递归"""
    if k == 0:
        return 1
    if n == 0:
        return 0
    
    return combinations_optimized(n - 1, k) + combinations_optimized(n, k - 1)

# 这就是面试官问的"怎么优化递归"的答案！
# lru_cache 自动缓存函数调用结果，避免重复计算
```
